const checkPowerOf2 = function (n) {
    var i = 0
    // 一步步增加2的幂次 并对比
    while (i < 32) {
        let multiple = Math.pow(2, i); // 当前幂次
        if (n === multiple) {
            return true // 是2的幂次
        }
        if (multiple > n) return false; // n小于当前幂次 return false 取消无 用遍历 缩短运行时间
        i++
    }
    return false
}
